Tabla 1: users
    id - Serial (PK)
    name - Text 
    surname - Text
    telephone - Integer
    address - Text

Tabla 2: travels
    id - Serial (PK)
    driver_id - Integer (AK)
    num_passengers - Integer
    date - Date
    start_time - Time
    finish_time - Time
    city - Text
    street - Text
    postalcode - Integer

Tabla 3: passengers
    id - Serial (PK)
    travel_id - Integer (AK)
    user_id - Integer (AK)

Tabla 4: categories
    id - Serial (PK)
    name - Text
